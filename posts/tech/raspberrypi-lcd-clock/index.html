#å£°æ˜ LCD pinsï¼ˆå¯¹åº”BCMå¼•è„šï¼‰

        lcd_d4, lcd_d5, lcd_d6, lcd_d7, lcd_columns, lcd_rows,\
        lcd_backlight)

while True:
    lcd.set_cursor(0,1)
    localtime = time.asctime( time.localtime(time.time()) )[4:-5]
    print(localtime)
    lcd.message(localtime)
    time.sleep(1)
